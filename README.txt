命令使用教程(v250815-164900)
================================================================================================
frp							# 内网穿透相关配置
	key							# 内网穿透访问密钥相关配置
		query						# 查询密钥
		set							# 设置密钥
		profile						# 相关配置存储
			save						# 保存配置
			load						# 加载配置
	start						# 启动内网穿透 (X)
		port						# 启动端口号 (X)
			[数字]						# 具体数字 (X)
	stop						# 停止内网穿透 (X)
================================================================================================
server						# 服务器 / 客户端操作
	download					# 下载服务器 / 客户端操作
		officialversion				# 下载服务器 / 客户端启动文件(xxx.jar)
			[具体版本]					# 具体版本
		manifest					# 下载服务器 / 客户端配置文件
			all							# 下载所有
			[具体版本]					# 下载具体版本
	create						# 创建服务器
		[服务器 / 客户端名称]			# 具体名称
	select						# 选择已创建的服务器
		[服务器 / 客户端名称]			# 具体名称
	unload						# 取消选择服务器
		[服务器 / 客户端名称]			# 具体名称
	list						# 列出已创建的服务器
		[服务器 / 客户端名称]			# 具体名称
	delete						# 删除已创建的服务器
		[服务器 / 客户端名称]			# 具体名称
	java64						# java64 启动配置
		select						# 选择已存在的64位java
			[已存在的java版本]				# 具体版本
	make						# 创建服务器启动文件
	core						# 服务器核心选择 (X)
		[core]						# 具体核心 (X)
	argument					# 相关启动参数
		memory						# 内存相关
			maxgb						# 最大内存(GB)
				[数字]						# 具体数字
			maxmb						# 最大内存(MB)
				[数字]						# 具体数字
			mingb						# 最小内存(GB)
				[数字]						# 具体数字
			minmb						# 最小内存(MB)
				[数字]						# 具体数字
			query						# 查询内存并给出推荐
	manifest				# 总版本配置文件操作
		update					# 总版本配置文件更新
================================================================================================
java						# 计算机已安装java操作
	list						# 列出java版本
	update						# 查找电脑已安装的java版本
================================================================================================
download					# 下载
	[链接]						# 具体链接
================================================================================================
quit						# 退出
